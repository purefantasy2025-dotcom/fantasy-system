from flask import Blueprint from flask import render_template from app.models import Customer from app.models import Product from app.models import Payment from app import db dashboard_bp = Blueprint('dashboard_bp', __name__) @dashboard_bp.route('/') @dashboard_bp.route('/dashboard') def dashboard(): total_customers = Customer.query.count() total_revenue = db.session.query( db.func.sum(Payment.amount) ).scalar() or 0 total_stock = db.session.query( db.func.sum(Product.stock) ).scalar() or 0 recent_payments = Payment.query.order_by( Payment.timestamp.desc() ).limit(5).all() return render_template( 'dashboard.html', total_customers=total_customers, total_revenue=total_revenue, total_stock=total_stock, recent_payments=recent_payments )
